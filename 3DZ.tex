\item Odredite rang eliptičke krivulje nad $\mathbb{Q}$ zadane jednadžbom
\[y^2=x^3-22x\]
\item Za polinom
\[p(x) = (x - 4)(x - 3)(x - 2)x(x + 1)(x + 2)(x + 3)(x + 4),\]
odredite polinome $q(x), r(x) \in \mathbb{Q}[x]$ takve da vrijedi $p(x)=(q(x))^2-r(x)$ i $\deg r \leq 3$
\end{enumerate}
